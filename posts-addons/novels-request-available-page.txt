    <script>
        // Replace with your Google Apps Script Web app URL
        const scriptURL = "https://script.google.com/macros/s/AKfycbyv-boesbCqIjdkBcPVeJXGp_A9fNWtjMSourReTlV7sWZmfK72O4eK8y1SKMek-hV1/exec";

        function isOnline() {
            return navigator.onLine;
        }

        function generateUUID() {
            if (crypto && crypto.randomUUID) {
                return crypto.randomUUID();
            }
            return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
                const r = Math.random() * 16 | 0, v = c === 'x' ? r : (r & 0x3 | 0x8);
                return v.toString(16);
            });
        }

        function submitRequest(event) {
            event.preventDefault();
            const submitButton = document.getElementById("submitButton");
            submitButton.disabled = true;
            submitButton.textContent = "Submitting...";

            const novelName = document.getElementById("novelName").value.trim();
            const authorName = document.getElementById("authorName").value.trim();
            const userId = localStorage.getItem("userId") || generateUUID();
            localStorage.setItem("userId", userId);
            const deviceType = navigator.userAgent.includes("Mobile") ? "Mobile" : "Desktop";
            const successMessage = document.getElementById("successMessage") || document.createElement("p");
            const errorMessage = document.getElementById("errorMessage") || document.createElement("p");

            // Reset messages
            document.querySelectorAll(".error").forEach(el => el.style.display = "none");
            if (!document.getElementById("successMessage")) {
                successMessage.id = "successMessage";
                successMessage.className = "success-message";
                document.getElementById("requestSection").appendChild(successMessage);
            }
            if (!document.getElementById("errorMessage")) {
                errorMessage.id = "errorMessage";
                errorMessage.className = "error-message";
                document.getElementById("requestSection").appendChild(errorMessage);
            }
            successMessage.textContent = "";
            errorMessage.textContent = "";

            // Validate inputs
            let isValid = true;
            if (!novelName) {
                document.getElementById("novelNameError").style.display = "block";
                isValid = false;
            }
            if (!authorName) {
                document.getElementById("authorNameError").style.display = "block";
                isValid = false;
            }

            if (!isValid) {
                submitButton.disabled = false;
                submitButton.textContent = "Submit Request";
                return;
            }

            if (!isOnline()) {
                errorMessage.textContent = "No internet connection. Please check your connection and try again.";
                submitButton.disabled = false;
                submitButton.textContent = "Submit Request";
                return;
            }

            const data = `novelName=${encodeURIComponent(novelName)}&authorName=${encodeURIComponent(authorName)}&userId=${encodeURIComponent(userId)}&deviceType=${encodeURIComponent(deviceType)}`;

            fetch(scriptURL, {
                method: "POST",
                headers: { "Content-Type": "application/x-www-form-urlencoded" },
                body: data,
            })
                .then(response => response.json())
                .then(data => {
                    if (data.status === "success") {
                        successMessage.textContent = "Your request has been submitted successfully!";
                        document.getElementById("requestForm").reset();
                    } else {
                        errorMessage.textContent = data.message || "Failed to submit request. Please try again.";
                    }
                })
                .catch(error => {
                    errorMessage.textContent = "Failed to submit request. Please try again.";
                })
                .finally(() => {
                    submitButton.disabled = false;
                    submitButton.textContent = "Submit Request";
                });
        }

        // Event Listener
        document.getElementById("requestForm").addEventListener("submit", submitRequest);
    </script>
